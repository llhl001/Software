Query {
    本地库就一个查个鬼
}

Create {
    shell> git init
    shell> git clone
}

Delete {
    直接删除本地仓库中的 .git 文件夹 即可，删除之后，该仓库就是普通文件夹，没有版本控制功能了
}

Update {
    shell> git remote add --mirror=fetch origin url         # 将 RemoteRepo 覆盖到 LocalRepo，该命令只对 BareRepo 有效
    本地 commit   # 查看额外的文件
    本地 branch   # 查看额外的文件 
    
}
