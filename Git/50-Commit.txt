}

}

Rollback {
    shell> git reset --hard HEAD^               # HEAD^ 上个版本， HEAD^^ 上上个版本， HEAD~100 上100个版本
    shell> git reset --hard [commit id]         # 指定回到未来某个版本 
    shell> git reset --hard origin/master       # 回到 origin 远程库的 master 分支的 HEAD 上
    
    shell> git revert 
}
