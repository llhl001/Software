通过 OpenSSL 自建 CA 认证机构 {
    创建 CA 根目录 {
        shell> mkdir ~/ca
        shell> cd ~/ca
    }
    
    生成 CA 认证机构根证书（CA 根证书都是 自签名 证书） {
        shell> openssl ecparam -list_curves             # 列出所有支持的曲线，选择自己浏览器支持的曲线，来生成密钥
        shell> openssl ecparam -genkey -name prime256v1 -out ./rootca.key.pem       # 生成 CA 根证书密钥对，包含公私钥
        shell> openssl pkey -pubout -in ./rootca.key.pem -out ./rootca.pub.pem      # 从密钥对中拆分出公钥到独立的文件中
        shell> touch myOpenssl.cnf          # 创建一个自己的 openssl.cnf 配置文件，可以使用 Software/OpenSSL/openssl.cnf 作为模板修改
        shell> openssl req -new -x509 -config ./myOpenssl.cnf -key ./rootca.key.pem -out ./rootca.cert.pem  # 生成 CA 自签名根证书
    }
    
    生成 Nginx 证书 {
        shell> openssl ecparam -genkey -name prime256v1 -out ./nginx.key.pem
        shell> openssl req -new -config ./myOpenssl.cnf -key ./nginx.key.pem -out ./nginx.csr         # 生成证书请求文件
    }
    
    使用 CA 证书签署 Nginx 证书 {
        shell> openssl x509 -req -in ./nginx.csr -CA ./rootca.cert.pem -CAkey ./rootca.key.pem -out ./nginx.crt

    }
}
